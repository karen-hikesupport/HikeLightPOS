        public void Stop()
        {

            if (_spi != null)
            {

                _spi.TxFlowStateChanged -= OnTxFlowStateChanged; // Called throughout to transaction process to update us with progress
                _spi = null;
            }
        }

        // Called during the pairing process to let us know how it's going.
        // We just update our screen with the information, and provide relevant Actions to the user.
        private void OnPairingFlowStateChanged(object sender, PairingFlowState pairingFlowState)
        {
            PairingFlowStateChanged?.Invoke(sender, pairingFlowState);
            Debug.WriteLine("# --------- PAIRING FLOW UPDATE -----------");
            Debug.WriteLine("# Message: {0}", pairingFlowState.Message);
            if (!string.IsNullOrEmpty(pairingFlowState.ConfirmationCode))
            {
                Debug.WriteLine("# Confirmation Code: {0}", pairingFlowState.ConfirmationCode);
            }
            PrintStatusAndActions();
        }


        /// Called during a transaction to let us know how it's going.
        /// We just update our screen with the information, and provide relevant Actions to the user.
        private void OnTxFlowStateChanged(object sender, TransactionFlowState txFlowState)
        {
            TransactionFlowStateChanged?.Invoke(sender, txFlowState);
            Debug.WriteLine("# --------- TRANSACTION FLOW UPDATE -----------");
            Debug.WriteLine("# Id: {0}", txFlowState.Id);
            Debug.WriteLine("# Type: {0}", txFlowState.Type);
            Debug.WriteLine("# RequestSent: {0}", txFlowState.RequestSent);
            Debug.WriteLine("# WaitingForSignature: {0}", txFlowState.AwaitingSignatureCheck);
            Debug.WriteLine("# Attempting to Cancel : {0}", txFlowState.AttemptingToCancel);
            Debug.WriteLine("# Finished: {0}", txFlowState.Finished);
            Debug.WriteLine("# Success: {0}", txFlowState.Success);
            Debug.WriteLine("# Display Message: {0}", txFlowState.DisplayMessage);

            if (txFlowState.AwaitingSignatureCheck)
            {
                // We need to print the receipt for the customer to sign.
                Debug.WriteLine(txFlowState.SignatureRequiredMessage.GetMerchantReceipt().TrimEnd());
            }

            // If the transaction is finished, we take some extra steps.
            if (txFlowState.Finished)
            {
                if (txFlowState.Success == Message.SuccessState.Unknown)
                {
                    // TH-4T, TH-4N, TH-2T - This is the dge case when we can't be sure what happened to the transaction.
                    // Invite the merchant to look at the last transaction on the EFTPOS using the dicumented shortcuts.
                    // Now offer your merchant user the options to:
                    // A. Retry the transaction from scrtatch or pay using a different method - If Merchant is confident that tx didn't go through.
                    // B. Override Order as Paid in you POS - If Merchant is confident that payment went through.
                    // C. Cancel out of the order all together - If the customer has left / given up without paying 
                    Debug.WriteLine("# NOT SURE IF WE GOT PAID OR NOT. CHECK LAST TRANSACTION MANUALLY ON EFTPOS!");
                }
                else
                {
                    // We have a result...
                    switch (txFlowState.Type)
                    {
                        // Depending on what type of transaction it was, we might act diffeently or use different data.
                        case TransactionType.Purchase:
                            if (txFlowState.Response != null)
                            {
                                var purchaseResponse = new PurchaseResponse(txFlowState.Response);
                                Debug.WriteLine("# Scheme: {0}", purchaseResponse.SchemeName);
                                Debug.WriteLine("# Response: {0}", purchaseResponse.GetResponseText());
                                Debug.WriteLine("# RRN: {0}", purchaseResponse.GetRRN());
                                Debug.WriteLine("# Error: {0}", txFlowState.Response.GetError());
                                Debug.WriteLine("# Customer Receipt:");
                                Debug.WriteLine(purchaseResponse.GetCustomerReceipt().TrimEnd());
                            }
                            else
                            {
                                // We did not even get a response, like in the case of a time-out.
                            }
                            if (txFlowState.Success == Message.SuccessState.Success)
                            {
                                // TH-6A
                                Debug.WriteLine("# HOORAY WE GOT PAID (TH-7A). CLOSE THE ORDER!");
                            }
                            else
                            {
                                // TH-6E
                                Debug.WriteLine("# WE DIDN'T GET PAID. RETRY PAYMENT (TH-5R) OR GIVE UP (TH-5C)!");
                            }
                            break;
                        case TransactionType.Refund:
                            if (txFlowState.Response != null)
                            {
                                var refundResponse = new RefundResponse(txFlowState.Response);
                                Debug.WriteLine("# Scheme: {0}", refundResponse.SchemeName);
                                Debug.WriteLine("# Response: {0}", refundResponse.GetResponseText());
                                Debug.WriteLine("# RRN: {0}", refundResponse.GetRRN());
                                Debug.WriteLine("# Error: {0}", txFlowState.Response.GetError());
                                Debug.WriteLine("# Customer Receipt:");
                                Debug.WriteLine(refundResponse.GetCustomerReceipt().TrimEnd());
                            }
                            else
                            {
                                // We did not even get a response, like in the case of a time-out.
                            }
                            break;
                        case TransactionType.Settle:
                            if (txFlowState.Response != null)
                            {
                                var settleResponse = new Settlement(txFlowState.Response);
                                Debug.WriteLine("# Response: {0}", settleResponse.GetResponseText());
                                Debug.WriteLine("# Error: {0}", txFlowState.Response.GetError());
                                Debug.WriteLine("# Merchant Receipt:");
                                Debug.WriteLine(settleResponse.GetReceipt().TrimEnd());
                            }
                            else
                            {
                                // We did not even get a response, like in the case of a time-out.
                            }
                            break;
                        default:
                            break;
                    }
                }
            }

            // Let's show the user what options he has at this stage.
            PrintStatusAndActions();
        }


        /// Called when Secrets are set or changed or voided.
        /// </summary>
        /// <param name="sender"></param>
        /// <param name="newSecrets"></param>
        private void OnSecretsChanged(object sender, Secrets newSecrets)
        {
            spiSecrets = newSecrets;
            if (spiSecrets != null)
            {
                Debug.WriteLine("\n\n\n# --------- I GOT NEW SECRETS -----------");
                Debug.WriteLine("# ---------- PERSIST THEM SAFELY ----------");
                Debug.WriteLine("# {0}:{1}", spiSecrets.EncKey, spiSecrets.HmacKey);
                Debug.WriteLine("# -----------------------------------------");
            }
            else
            {
                Debug.WriteLine("\n\n\n# --------- THE SECRETS HAVE BEEN VOIDED -----------");
                Debug.WriteLine("# ---------- CONSIDER ME UNPAIRED ----------");
                Debug.WriteLine("# -----------------------------------------");
            }
        }

        /// <summary>
        /// This method prints the current Spi Status and Flow
        /// and lists available actions to the user.
        /// </summary>
        private void PrintStatusAndActions()
        {
            Debug.WriteLine("# ----------- AVAILABLE ACTIONS ------------");

            // Available Actions depend on the current status (Unpaired/PairedConnecting/PairedConnected)
            switch (_spi.CurrentStatus)
            {
                case SpiStatus.Unpaired: //Unpaired...
                    switch (_spi.CurrentFlow)
                    {
                        case SpiFlow.Idle: // Unpaired, Idle
                            Debug.WriteLine("# [pos_id:MYPOSNAME] - sets your POS instance ID");
                            Debug.WriteLine("# [eftpos_address:10.10.10.10] - sets IP address of target EFTPOS");
                            Debug.WriteLine("# [pair] - start pairing");
                            break;

                        case SpiFlow.Pairing: // Unpaired, PairingFlow
                            var pairingState = _spi.CurrentPairingFlowState;
                            if (pairingState.AwaitingCheckFromPos)
                            {
                                ProcessStatusChanged?.Invoke(this, "Pair confirm - Confirm the code matches");
                            }
                            if (!pairingState.Finished)
                            {
                                ProcessStatusChanged?.Invoke(this, "Pair cancel - cancel pairing process");
                            }
                            else
                            {
                                Debug.WriteLine("# [ok] - acknowledge");
                            }
                            break;

                        case SpiFlow.Transaction: // Unpaired, TransactionFlow - Should never be the case!
                        default:
                            ProcessStatusChanged?.Invoke(this, "# .. Unexpected Flow .. "+ _spi.CurrentFlow);
                            break;
                    }
                    break;

                case SpiStatus.PairedConnected:
                    switch (_spi.CurrentFlow)
                    {
                        case SpiFlow.Idle: // Paired, Idle
                            ProcessStatusChanged?.Invoke(this, "Device is Idle");
                            break;
                        case SpiFlow.Transaction: // Paired, Transaction
                            if (_spi.CurrentTxFlowState.AwaitingSignatureCheck)
                            {
                                ProcessStatusChanged?.Invoke(this, "Transaction sign accept - Accept Signature");
                                ProcessStatusChanged?.Invoke(this, "Transaction sign decline - Decline Signature");
                            }
                            if (!_spi.CurrentTxFlowState.Finished)
                            {
                                ProcessStatusChanged?.Invoke(this, "Transaction cancel - Attempt to Cancel Transaction");
                            }else
                            {
                                ProcessStatusChanged?.Invoke(this, "ok - acknowledge");
                            }
                            break;
                        case SpiFlow.Pairing: // Paired, Pairing - we have just finished the pairing flow. OK to ack.
                            ProcessStatusChanged?.Invoke(this, "ok - acknowledge");
                            break;
                        default:
                            ProcessStatusChanged?.Invoke(this, "Unexpected Flow .. " + _spi.CurrentFlow);
                            break;
                    }
                    break;

                case SpiStatus.PairedConnecting: // This is still considered as a Paired kind of state, but...
                    goto case SpiStatus.PairedConnected;

                default:
                    ProcessStatusChanged?.Invoke(this, "Unexpected State .." + _spi.CurrentStatus);
                    break;
            }
            ProcessStatusChanged?.Invoke(this, "Status :" + _spi.CurrentStatus + "  and Flow is:" + _spi.CurrentFlow);
        }

        public bool PairDevice()

        public void CancelPairing()

        public bool UnpairDevice()
        {
            return _spi.Unpair();

        public InitiateTxResult Purchase(string posRefId, int amount)
        {
            var pres = _spi.InitiatePurchaseTx(posRefId, amount);
            if (pres.Initiated)
            {
                ProcessStatusChanged?.Invoke(this, "Purchase Initiated. Will be updated with Progress.");
            }
            else
            {
                ProcessStatusChanged?.Invoke(this, "Could not initiate purchase:" + pres.Message + ".Please Retry.");
            }
            return pres;

        }

        public InitiateTxResult Refund(string posRefId, int amount)
        {
            var rres = _spi.InitiateRefundTx(posRefId, amount);
            if (rres.Initiated)
            {
                ProcessStatusChanged?.Invoke(this, "Refund Initiated. Will be updated with Progress.");
            }
            else
            {
                ProcessStatusChanged?.Invoke(this, "Could not initiate refund: " + rres.Message + ".Please Retry.");
            }
            return rres;
        }

        public InitiateTxResult Settle(string posRefId)
        {
            var sres = _spi.InitiateSettleTx(posRefId);
            if (sres.Initiated)
            {
                ProcessStatusChanged?.Invoke(this, "Settle Initiated. Will be updated with Progress.");
            }
            else
            {
                ProcessStatusChanged?.Invoke(this, "Could not initiate settle: " + sres.Message + ". Please Retry.");
            }
            return sres;
        }

        public MidTxResult AcceptSignature(bool accept)
        {
            return _spi.AcceptSignature(accept);
        }

        public MidTxResult CancelTransaction()
        {
            return _spi.CancelTransaction();
        }

        public bool AckFlowEndedAndBackToIdle()
        {
            var result = _spi.AckFlowEndedAndBackToIdle();
            PrintStatusAndActions();
            return result;
        }
